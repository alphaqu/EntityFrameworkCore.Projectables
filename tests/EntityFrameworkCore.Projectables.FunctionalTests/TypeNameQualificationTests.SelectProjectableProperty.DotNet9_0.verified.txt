SELECT [e2].[Id], [e2].[EntityId], [e2].[ParentId]
FROM [Entity] AS [e]
LEFT JOIN (
    SELECT [e1].[Id], [e1].[EntityId], [e1].[ParentId]
    FROM (
        SELECT [e0].[Id], [e0].[EntityId], [e0].[ParentId], ROW_NUMBER() OVER(PARTITION BY [e0].[EntityId] ORDER BY [e0].[Id]) AS [row]
        FROM [Entity] AS [e0]
    ) AS [e1]
    WHERE [e1].[row] <= 1
) AS [e2] ON [e].[Id] = [e2].[EntityId]
